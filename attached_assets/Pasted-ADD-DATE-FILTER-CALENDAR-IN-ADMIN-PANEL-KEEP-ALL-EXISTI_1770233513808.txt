3. BACKEND - ADD NEW API ROUTE:

CREATE a new route (or modify existing route) to accept date parameter:
```javascript
// ADD this new route in your bookings routes file
// Route: GET /api/bookings/by-date?date=2026-02-05

router.get('/bookings/by-date', async (req, res) => {
  try {
    const { date } = req.query; // Format: YYYY-MM-DD
    
    // If no date provided, use today
    const targetDate = date ? new Date(date) : new Date();
    
    // Create date range for the entire day
    const startOfDay = new Date(targetDate);
    startOfDay.setHours(0, 0, 0, 0);
    
    const endOfDay = new Date(targetDate);
    endOfDay.setHours(23, 59, 59, 999);
    
    // Find all WAITING bookings for this date
    const bookings = await Booking.find({
      status: 'WAITING',
      createdAt: {
        $gte: startOfDay,
        $lte: endOfDay
      }
    })
    .sort({ createdAt: 1 }) // Sort by oldest first
    .exec();
    
    // Add sequential queue position numbers
    const bookingsWithPosition = bookings.map((booking, index) => ({
      ...booking.toObject(),
      queuePosition: index + 1  // #1, #2, #3, #4...
    }));
    
    // Return bookings with count
    res.json({
      bookings: bookingsWithPosition,
      totalWaiting: bookingsWithPosition.length,
      date: targetDate.toISOString().split('T')[0]
    });
    
  } catch (error) {
    console.error('Error fetching bookings by date:', error);
    res.status(500).json({ error: 'Failed to fetch bookings' });
  }
});
```
