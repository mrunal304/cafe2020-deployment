### Step 2: Update Home Page Component

Find `client/src/pages/Home.tsx` (or wherever logo is displayed):

```jsx
import { useEffect, useState } from 'react';
import cafeLogo from '../assets/cafe-logo.png';

export default function Home() {
  // Animation states
  const [showLogo, setShowLogo] = useState(false);
  const [showTitle, setShowTitle] = useState(false);
  const [showTagline, setShowTagline] = useState(false);

  // Trigger animations on mount
  useEffect(() => {
    setTimeout(() => setShowLogo(true), 100);
    setTimeout(() => setShowTitle(true), 500);
    setTimeout(() => setShowTagline(true), 900);
  }, []);

  return (
    <div className="min-h-screen bg-gradient-to-br from-[#F5EDE4] via-[#E8DDD0] to-[#DED0BF]">
      
      {/* Logo Section with Animations */}
      <div className="flex flex-col items-center justify-center pt-12 pb-8 px-6">
        
        {/* Logo - Larger with Fade & Scale Animation */}
        <img 
          src={cafeLogo}
          alt="Cafe Twenty Twenty"
          className={`
            w-56 h-56 object-contain mb-6
            transition-all duration-1000 ease-out
            ${showLogo 
              ? 'opacity-100 translate-y-0 scale-100' 
              : 'opacity-0 -translate-y-8 scale-90'
            }
          `}
        />
        
        {/* Cafe Name - White, Serif Font, Fade In */}
        <h1 
          className={`
            text-5xl md:text-6xl font-bold text-white text-center mb-3
            transition-all duration-1000 ease-out
            ${showTitle 
              ? 'opacity-100 translate-y-0' 
              : 'opacity-0 translate-y-4'
            }
          `}
          style={{ fontFamily: "'Playfair Display', serif" }}
        >
          Cafe Twenty Twenty
        </h1>
        
        {/* Tagline - White, Italic Serif, Fade In */}
        <p 
          className={`
            text-xl md:text-2xl text-white/90 text-center italic font-light
            transition-all duration-1000 ease-out
            ${showTagline 
              ? 'opacity-100 translate-y-0' 
              : 'opacity-0 translate-y-4'
            }
          `}
          style={{ fontFamily: "'Playfair Display', serif" }}
        >
          Roasted Rich. Served Rustic.
        </p>
        
      </div>
      
      {/* Join the Queue Form - Keep as is */}
      <div className="max-w-md mx-auto p-6 bg-white/70 rounded-3xl backdrop-blur-sm">
        <h2 className="text-2xl font-bold text-stone-900 mb-2">
          Join the Queue
        </h2>
        <p className="text-stone-600 text-sm mb-6">
          Get in line from anywhere. We'll text you when your table is ready.
        </p>
        
        {/* ... rest of the form stays exactly the same ... */}
      </div>
      
    </div>
  );
}
```

## Animation Details

### Logo Animation
```jsx
className={`
  w-56 h-56 object-contain mb-6
  transition-all duration-1000 ease-out
  ${showLogo 
    ? 'opacity-100 translate-y-0 scale-100'    // Final state: visible, in place, normal size
    : 'opacity-0 -translate-y-8 scale-90'      // Initial state: invisible, above, slightly smaller
  }
`}
```

### Title Animation
```jsx
className={`
  text-5xl font-bold text-white
  transition-all duration-1000 ease-out
  ${showTitle 
    ? 'opacity-100 translate-y-0'   // Final: visible, in place
    : 'opacity-0 translate-y-4'     // Initial: invisible, slightly below
  }
`}
```

### Tagline Animation
```jsx
className={`
  text-xl text-white/90 italic
  transition-all duration-1000 ease-out
  ${showTagline 
    ? 'opacity-100 translate-y-0'   // Final: visible, in place
    : 'opacity-0 translate-y-4'     // Initial: invisible, slightly below
  }
`}
```

## Font Styling

### Cafe Twenty Twenty
- Font: Playfair Display (serif, elegant)
- Weight: Bold (700)
- Size: `text-5xl` (3rem) on mobile, `text-6xl` (3.75rem) on desktop
- Color: White

### Roasted Rich. Served Rustic.
- Font: Playfair Display Italic
- Weight: Light (400)
- Size: `text-xl` (1.25rem) on mobile, `text-2xl` (1.5rem) on desktop
- Color: White with 90% opacity (`text-white/90`)
